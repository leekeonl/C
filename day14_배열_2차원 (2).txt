#include <stdio.h>

void main() {


//-------------------------------------------------------------

	/*
	다차원 배열
	 1. 1차원 배열의 모임
	 
	2. 사람이 볼때는 행과 열로 구성된 구조이지만, 실제 메모리에 저장 될 때는  1차원의 순서로 저장한다.
	 
	 ex ) 	자료형	배열명	행  열
		  int	   n   	[3] [3]
		
	사람  =	(0,0) (0,1) (0,2)     
		(1,0) (1,1) (1,2)  
		(2,0) (2,1) (2,2) 
		
		-------------------------------------------------------------------------	
	메모리=	| (0,0) | (0,1) | (0,2) | (1,0) | (1,1) | (1,2) | (2,0) | (2,1) | (2,2) |	
		-------------------------------------------------------------------------
 	
	3. 2차원 배열은 대괄호 두 개를 사용하여 만들고 요소를 표현 할 때에도 대괄호 두개를  사용해야 한다.
	
	4. 배열을 만들 때에는 중괄호를 이용하여 초기ㄱ밧을 부여하는데, 초기값을 부여하는 경우  행의 개수만은 생략 가능하다.
		ex ) int a[2][3]={1,2,3,4,5,6};  or  int a[][3]={1,2,3,4,5,6};
	
	5. 초기값 부여시에 행 구분으로 중괄호를 사용 할 수 있다.
		
	   ex ) int a[2][3]={ {1,2,3} , {4,5,6} }; => 이 표현 방식을 권장 한다.
 
	*/
 	
	// 2차원 배열의 초기화 방법
	int  a[3][3] = { 11, 23, 38, 49, 52, 68, 73, 80, 99 };
	int  b[3][3] = { { 11, 23, 38 },  { 49, 52, 68 }, { 73, 80, 99 } }; // 권장
	int  b[3][3] = { { 11, 23, 38 },  { 49, 52, 68 },  73, 80, 99  };
	int  b[3][3] = { 11, 23, 38 ,  { 49, 52, 68 }, { 73, 80, 99 } };  
	
	// 배열 호출 형식을 이용하여  위 배열의 23을 출력
	printf(“b 배열의  23 호출  = %d  \n”, b[0][1]);
	
	// 68을 출력
	printf(“b 배열의  68 호출 = %d  \n”,  b[1][2] );

	
	// for문을  이용한 다차원 배열의 일괄  출력
	int a[3][3] = {	{ 11, 12, 13 }, { 21, 22, 23 },	{ 31, 32, 33 }};
	int  i, j ;
	
	for( i=0; i<3; ++i) {               
		for( j=0; j<3; ++j) { 
		      printf("  배열 요소 a[%d][%d] : %d", i, j, a[i][j]);
		}
		printf("\n") ;
	}

}























}
